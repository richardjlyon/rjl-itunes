def protected?
  protected = false
  @tracks.each do |track|
    if track.tags.include? 'protected'
      protected = true
      break
    end
  end
  return protected
end

def unique?( parameter_name )
  values = []
  @tracks.each do |track|
    values << track.send( parameter_name )
  end
  return values.uniq.length == 1
end
